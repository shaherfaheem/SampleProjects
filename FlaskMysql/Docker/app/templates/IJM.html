-- INNER JOINS MULTIPLE TABLES --

SELECT o.order_id, item_id, oi.item_name, oi.item_quantity, o.order_status, s.shipper_id, s.shipping_details

FROM order_items oi

JOIN orders o ON oi.order_id = o.order_id

JOIN shipping s ON s.shipper_id = o.shipper_id;
